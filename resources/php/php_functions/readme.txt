// login() => Name, Passwort
//
// Rückgabe:
//   6 = User und Password = NULL
//   5 = User = NULL
//   4 = Password = NULL
//   3 = User existiert nicht!
//   2 = Passwort ist falsch!
//   1 = Eingeloggt
// ------------------------------

// ------------------------------
// regist() => Name, Passwort, Email
//
// Rückgabe:
//  11 = User, Email und Passwort = NULL
//  10 = User und Email = NULL
//   9 = User und Passwort = NULL
//   8 = Passwort und Email = NULL
//   7 = User = NULL
//   6 = Password = NULL
//   5 = Email = NULL
//   4 = User und Email existieren!
//   3 = User existiert!
//   2 = Email existiert!
//   1 = Registriert
// ------------------------------
